# Alias
# Fedora
```
alias mysqlR="mysql -h localhost -u root -p"
alias mysqlH="mysql -h localhost -u hec -p"
alias mysqlstart="sudo systemctl start mariadb && sudo systemctl is-active mariadb.service"
alias mysqlstop="sudo systemctl stop mariadb && sudo systemctl is-active mariadb.service"
alias apachestart="sudo systemctl start httpd && sudo systemctl is-active httpd.service"
alias apachestop="sudo systemctl stop httpd && sudo systemctl is-active httpd.service"
alias py="python3"
alias cls="clear"
alias gitUpGitHubMain="git push -u origin main"
alias gitUpHerokuMain="git push heroku main"
alias prog="cd '/mnt/Datos/Prog'"
```
# Arch
```
alias mysql="/opt/lampp/bin/mysql -u root"
alias mysqlstart="sudo /opt/lampp/lampp startmysql"
alias mysqlstop="sudo /opt/lampp/lampp stopmysql"
alias apachestart="sudo sudo /opt/lampp/lampp startapache"
alias apachestop="sudo sudo /opt/lampp/lampp stopapache"
alias lampstat="sudo /opt/lampp/lampp start"
alias lampstop="sudo /opt/lampp/lampp stop"
alias py="python3"
alias cls="clear"
```
